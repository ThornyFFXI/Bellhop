            <!-- Any bag indices listed here will be treated as if you do not have access to them.  This is primarily for DSP. -->

            <forcedisablebags>9</forcedisablebags>

            <nomadstorage>Enabled/Disabled</nomadstorage>

            <ignorecraftskill>Enabled/Disabled</ignorecraftskill>

 

            <!--If enabled, commands used for Ashita3's version of bellhop will work.  This is disabled by default as some of them overlap with ingame commands or are generic enough to compete with other plugins. -->            <legacycommands>Enabled/Disabled</legacycommands>

 

            <!-- If enabled, complex commands will be summarized in one line of output instead of using multiple. -->

            <shortoutput>Enabled/Disabled</shortoutput>

 

<!-- The time, in milliseconds, before an autotrade-related packet will be resent if the server does not respond to it.  Must be between 1 and 9999. -->
